Interface CalendarConfigMonth

A definition of a month within a calendar year.

interface CalendarConfigMonth {
    abbreviation?: string;
    dayOffset?: number;
    days: number;
    intercalary?: boolean;
    leapDays?: number;
    name: string;
    ordinal: number;
    startingWeekday?: null | number;
}
